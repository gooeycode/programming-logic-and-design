readOpen()

writeLoop()
	if (tId = mId) {
		mTotalSpent = mTotalSpent + tServiceCost;
		output mFirstName, mLastName, mTotalSpent, mId to updatedFile;
		readMaster();
		readTransaction();
	}end if
	else if (tId > mI) {
		output mFirstName, mLastName, mTotalSpent, mId, to updatedFile;
		readMaster();
	}end else if
	else {
		output ERROR;
		output mFirstName, mLastName, mTotalSpent, mId to updatedFile;
		readTransaction();
	}end else
	checkEnd();
	return

finish();
